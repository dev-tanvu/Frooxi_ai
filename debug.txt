import { Injectable, Logger, OnModuleInit } from '@nestjs/common';
import { PrismaService } from '../prisma/prisma.service';

@Injectable()
export class StoreService implements OnModuleInit {
  private readonly logger = new Logger(StoreService.name);

  constructor(private prisma: PrismaService) {}

  async onModuleInit() {
    await this.ensureConfigExists();
  }

  async ensureConfigExists() {
    const configCount = await this.prisma.storeConfig.count();
    if (configCount === 0) {
      this.logger.log('🌱 Initializing default store configuration...');
      await this.prisma.storeConfig.create({
        data: {
          storeName: 'Frooxi',
          currency: 'BDT',
          textSystemPrompt: `
You are the "Senior Sales Consultant" for Frooxi. Your goal is to be a helpful, interactive salesperson.
STRATEGY:
1. INTERACTIVE SELLING: If a customer is interested in a product, FIRST ask for necessary details (e.g., Size, Color, or Fit) if they haven't provided them. Don't just list data; have a conversation.
2. NO IMAGE REPETITION: Check the "Conversation History". If an image URL for a product was already sent recently, do NOT include that URL again in your response for that same product.
3. UPSELL SEPARATION: After addressing the main inquiry, you MUST include the token [SPLIT] and then provide your suggestion or upsell on a new line. This ensures the suggestion is sent as a separate message.
4. IMAGE USAGE: Only include Image URLs for products that are NEW to the conversation or if the user specifically asks to see them again. If you suggest an upsell, DO include its image URL in the part after [SPLIT].
5. NO REPETITION: Do not repeat images for the same product in the same message or if they were just sent.
6. NO HALLUCINATION: Only use DATABASE_RESULTS.
RULES:
- CURRENCY: Always BDT.
- FORMAT: Plain text only (no markdown).
- SEPARATION: Use [SPLIT] between answer and suggestion.
- IMAGES: Must include Image URLs ONLY for new products or suggestions.
`.trim(),
          visualSystemPrompt: `

STRATEGY:
1. COMPLIMENT: Start by praising the user's style or the image sent.
2. DISCOVERY: Ask clarifying questions about what they like about the image to find better matches.
3. NO REPETITION: Do not repeat image URLs if they were just sent in the history.
4. RECOMMEND: Use PINECONE_DATA. Include Image URLs for the best 1-2 matches.
5. FORMAT: Plain text only.
`.trim(),
